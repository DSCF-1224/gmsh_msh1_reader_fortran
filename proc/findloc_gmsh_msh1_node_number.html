private elemental function findloc_gmsh_msh1_node_number(msh1_data, node_number) result(location)

Returns the location of the gmsh_msh1_node_type corresponding to the node_number argument.

Warning

If no gmsh_msh1_node_type corresponding to the node_number argument exists, zero will be returned.

    elemental function findloc_gmsh_msh1_node_number(msh1_data, node_number) result(location)

        type(gmsh_msh1_data_type), intent(in) :: msh1_data

        type(gmsh_msh1_node_number_type), intent(in) :: node_number

        integer :: location



        integer :: itr_node



        location = 0



        if ( .not. allocated(msh1_data%node) ) return



        do itr_node = 1, count_nodes(msh1_data)

            if ( msh1_data%node(itr_node)%node_number .eq. node_number ) then

                location = itr_node

                return

            end if

        end do

    end function findloc_gmsh_msh1_node_number
